1. Get the metrics URL:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "network-exporter.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl port-forward $POD_NAME 8000:8000 --namespace {{ .Release.Namespace }}

  Then access: http://localhost:8000/metrics

2. View logs:
  kubectl logs -f deployment/{{ include "network-exporter.fullname" . }} -n {{ .Release.Namespace }}

3. Check deployment status:
  kubectl get pods -n {{ .Release.Namespace }} | grep network-exporter
  kubectl describe pod -l app.kubernetes.io/name=network-exporter -n {{ .Release.Namespace }}

4. Active targets in this deployment:
  - 8.8.8.8 (ICMP - Google DNS)
  - 1.1.1.1 (ICMP - Cloudflare DNS)
  - 208.67.222.222 (ICMP - OpenDNS)
  - example.com:80 (HTTP)
  - google.com:443 (HTTPS)
  - cloudflare.com (DNS resolution via 1.1.1.1)
  - example.com:80 (TCP)
  - 8.8.8.8:53 (UDP)
  - pool.ntp.org (NTP)

5. Verify metrics are being collected:
  kubectl exec -it $(kubectl get pod -l app.kubernetes.io/name=network-exporter -n {{ .Release.Namespace }} -o jsonpath='{.items[0].metadata.name}') -n {{ .Release.Namespace }} -- curl http://localhost:8000/metrics | head -20

6. Add to Prometheus scrape config:
  scrape_configs:
    - job_name: 'network-exporter'
      kubernetes_sd_configs:
        - role: pod
          namespaces:
            names:
              - {{ .Release.Namespace }}
      relabel_configs:
        - source_labels: [__meta_kubernetes_pod_label_app_kubernetes_io_name]
          action: keep
          regex: network-exporter

7. Common commands:
  helm status network-exporter -n {{ .Release.Namespace }}
  helm get values network-exporter -n {{ .Release.Namespace }}
  helm upgrade network-exporter ./network-exporter-helm/network-exporter -n {{ .Release.Namespace }}
  helm rollback network-exporter 1 -n {{ .Release.Namespace }}
